int  Params_read( Params* m, FILE* f )
{
  int    e;
  int    i;
  TCHAR  line[256];

  line[0] = '\0';
  _fgetts( line, _countof(line), f );

  e= StrT_cutLineComment( line, sizeof(line), line, _T("//") ); IF(e)goto fin;
  m->BufferSize = _ttoi( line );

  _fgetts( line, _countof(line), f );  // skip space line
  e=0;
fin:
  return  e;
}
